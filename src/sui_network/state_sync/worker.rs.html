31pub(crate) fn build_object_store(
32    ingestion_url: &str,
33    remote_store_options: Vec<(String, String)>,
34) -> Arc<dyn ObjectStore> {
35    let timeout_secs = 5;
36    let client_options = ClientOptions::new()
37        .with_timeout(Duration::from_secs(timeout_secs))
38        .with_allow_http(true);
39    let retry_config = RetryConfig {
40        max_retries: 10,
41        retry_timeout: Duration::from_secs(timeout_secs + 1),
42        ..Default::default()
43    };
44    let url = ingestion_url
45        .parse::<url::Url>()
46        .expect("archival ingestion url must be valid");
47    if url.scheme() == "file" {
48        Arc::new(
49            LocalFileSystem::new_with_prefix(
50                url.to_file_path()
51                    .expect("archival ingestion url must have a valid file path"),
52            )
53            .expect("failed to create local file system store"),
54        )
55    } else if url.host_str().unwrap_or_default().starts_with("s3") {
56        let mut builder = AmazonS3Builder::new()
57            .with_client_options(client_options)
58            .with_retry(retry_config)
59            .with_imdsv1_fallback()
60            .with_url(ingestion_url);
61        for (key, value) in &remote_store_options {
62            builder = builder.with_config(
63                AmazonS3ConfigKey::from_str(key).expect("invalid S3 config key"),
64                value.clone(),
65            );
66        }
67        Arc::new(builder.build().expect("failed to build S3 store"))
68    } else {
69        Arc::new(
70            HttpBuilder::new()
71                .with_url(url.to_string())
72                .with_client_options(client_options)
73                .with_retry(retry_config)
74                .build()
75                .expect("failed to build HTTP store"),
76        )
77    }
78}

80pub(crate) async fn fetch_checkpoint(
81    store: &Arc<dyn ObjectStore>,
82    seq: u64,
83) -> anyhow::Result<Checkpoint> {
84    let store = store.clone();
85    let request = move || {
86        let store = store.clone();
87        async move {
88            use backoff::Error as BE;
89            let path = ObjectPath::from(format!("{seq}.binpb.zst"));
90            let bytes = store
91                .get(&path)
92                .await
93                .map_err(|e| match e {
94                    object_store::Error::NotFound { .. } => {
95                        BE::permanent(anyhow!("Checkpoint {seq} not found in archive"))
96                    }
97                    e => BE::transient(anyhow::Error::from(e)),
98                })?
99                .bytes()
100                .await
101                .map_err(|e| BE::transient(anyhow::Error::from(e)))?;
102            let decompressed =
103                zstd::decode_all(&bytes[..]).map_err(|e| BE::transient(anyhow::Error::from(e)))?;
104            let proto_checkpoint = proto::Checkpoint::decode(&decompressed[..])
105                .map_err(|e| BE::transient(anyhow::Error::from(e)))?;
106            Checkpoint::try_from(&proto_checkpoint).map_err(|e| BE::transient(anyhow!(e)))
107        }
108    };
109    let backoff = ExponentialBackoff {
110        max_elapsed_time: Some(Duration::from_secs(60)),
111        multiplier: 1.0,
112        ..Default::default()
113    };
114    backoff::future::retry(backoff, request).await
115}

117pub(crate) fn process_archive_checkpoint<S>(
118    store: &S,
119    checkpoint: &Checkpoint,
120    metrics: &Metrics,
121) -> anyhow::Result<()>
122where
123    S: WriteStore + Clone,
124{
125    let verified_checkpoint =
126        get_or_insert_verified_checkpoint(store, checkpoint.summary.clone(), true)?;
127    let full_contents = VersionedFullCheckpointContents::from_contents_and_execution_data(
128        checkpoint.contents.clone(),
129        checkpoint.transactions.iter().map(|t| ExecutionData {
130            transaction: Transaction::from_generic_sig_data(
131                t.transaction.clone(),
132                t.signatures.clone(),
133            ),
134            effects: t.effects.clone(),
135        }),
136    );
137    full_contents.verify_digests(verified_checkpoint.content_digest)?;
138    let verified_contents = VerifiedCheckpointContents::new_unchecked(full_contents);
139    store.insert_checkpoint_contents(&verified_checkpoint, verified_contents)?;
140    store.update_highest_synced_checkpoint(&verified_checkpoint)?;
141    metrics.update_checkpoints_synced_from_archive();
142    Ok(())
143}

145pub fn get_or_insert_verified_checkpoint<S>(
146    store: &S,
147    certified_checkpoint: CertifiedCheckpointSummary,
148    verify: bool,
149) -> anyhow::Result<VerifiedCheckpoint>
150where
151    S: WriteStore + Clone,
152{
153    store
154        .get_checkpoint_by_sequence_number(certified_checkpoint.sequence_number)
155        .map(Ok::<VerifiedCheckpoint, anyhow::Error>)
156        .unwrap_or_else(|| {
157            let verified_checkpoint = if verify {
158                // Verify checkpoint summary
159                let prev_checkpoint_seq_num = certified_checkpoint
160                    .sequence_number
161                    .checked_sub(1)
162                    .context("Checkpoint seq num underflow")?;
163                let prev_checkpoint = store
164                    .get_checkpoint_by_sequence_number(prev_checkpoint_seq_num)
165                    .context(format!(
166                        "Missing previous checkpoint {} in store",
167                        prev_checkpoint_seq_num
168                    ))?;
169
170                verify_checkpoint(&prev_checkpoint, store, certified_checkpoint)
171                    .map_err(|_| anyhow!("Checkpoint verification failed"))?
172            } else {
173                VerifiedCheckpoint::new_unchecked(certified_checkpoint)
174            };
175            // Insert checkpoint summary
176            store
177                .insert_checkpoint(&verified_checkpoint)
178                .map_err(|e| anyhow!("Failed to insert checkpoint: {e}"))?;
179            // Update highest verified checkpoint watermark
180            store
181                .update_highest_verified_checkpoint(&verified_checkpoint)
182                .expect("store operation should not fail");
183            Ok::<VerifiedCheckpoint, anyhow::Error>(verified_checkpoint)
184        })
185        .map_err(|e| anyhow!("Failed to get verified checkpoint: {:?}", e))
186}
